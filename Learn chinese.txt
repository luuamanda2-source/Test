    <script>
        const SCRIPT_URL = 'https://script.google.com/macros/s/AKfycbwo3gzi4yonrUXIpsaaHYfks6omtaZid6OKwmnxhZEOgttu_bA2Do1k3OhGj3sBT9CRmg/exec'; // Replace with your Google Apps Script URL
        
        async function loadExercisesFromSheet() {
            try {
                const response = await fetch(SCRIPT_URL);
                const data = await response.json();
                // Process exercises data
                console.log('Loaded exercises:', data);
            } catch (error) {
                console.error('Error loading exercises:', error);
            }
        }

        document.addEventListener('DOMContentLoaded', function() {
            const generateBtn = document.getElementById('generateBtn');
            const checkBtn = document.getElementById('checkBtn');
            const originalText = document.getElementById('originalText');
            const newWords = document.getElementById('newWords');
            const exerciseContent = document.getElementById('exerciseContent');
            const resultContainer = document.getElementById('resultContainer');
            
            let blanks = [];
            let correctAnswers = [];
            
            generateBtn.addEventListener('click', function() {
                const text = originalText.value.trim();
                const words = newWords.value.split(',').map(word => word.trim()).filter(word => word);
                
                if (!text) {
                    alert('Vui lòng nhập đoạn văn');
                    return;
                }
                
                if (words.length === 0) {
                    alert('Vui lòng nhập ít nhất một từ mới');
                    return;
                }
                
                // Create exercise content with blanks
                blanks = [];
                correctAnswers = [];
                let exerciseText = text;
                
                // Replace each word with a blank
                words.forEach(word => {
                    const regex = new RegExp(word, 'g');
                    exerciseText = exerciseText.replace(regex, `<input type="text" class="blank" data-answer="${word}">`);
                });
                
                // Store correct answers
                words.forEach(word => {
                    const regex = new RegExp(word, 'g');
                    const matches = text.match(regex);
                    if (matches) {
                        matches.forEach(() => correctAnswers.push(word));
                    }
                });
                
                exerciseContent.innerHTML = exerciseText;
                checkBtn.disabled = false;
                resultContainer.style.display = 'none';
                // Save to Google Sheet
                saveExerciseToSheet(text, words.join(', '));
            });

            async function saveExerciseToSheet(text, words) {
                try {
                    const response = await fetch(SCRIPT_URL, {
                        method: 'POST',
                        headers: {
                            'Content-Type': 'application/json',
                        },
                        body: JSON.stringify({ text, words }),
                    });
                    const result = await response.json();
                    console.log('Saved to sheet:', result);
                } catch (error) {
                    console.error('Error saving:', error);
                }
            }
            
            checkBtn.addEventListener('click', function() {
                const blankInputs = document.querySelectorAll('.blank');
                let allCorrect = true;
                let incorrectCount = 0;
                
                blankInputs.forEach((input, index) => {
                    const userAnswer = input.value.trim();
                    const correctAnswer = correctAnswers[index];
                    
                    if (userAnswer === correctAnswer) {
                        input.style.border = '2px solid #4CAF50';
                        input.style.backgroundColor = '#e8f5e9';
                    } else {
                        input.style.border = '2px solid #f44336';
                        input.style.backgroundColor = '#ffebee';
                        allCorrect = false;
                        incorrectCount++;
                    }
                });
                
                if (allCorrect) {
                    resultContainer.className = 'result correct';
                    resultContainer.innerHTML = '<strong>Chính xác!</strong> Bạn đã điền đúng tất cả các từ.';
                } else {
                    resultContainer.className = 'result incorrect';
                    resultContainer.innerHTML = `
                        <strong>Cần cải thiện!</strong> Bạn đã sai ${incorrectCount}/${correctAnswers.length} từ. 
                        <div>Đáp án đúng là: ${correctAnswers.join(', ')}</div>
                    `;
                }
                
                resultContainer.style.display = 'block';
            });
        });
